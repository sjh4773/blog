# t검정 - 두 집단의 평균 비교

't 검정(t-test)'은 두 집단의 평균에 통계적으로 유의한 차이가 있는지 알아볼 때 사용하는 통계 분석 기법

```r



# 일반 휘발유와 고급 휘발유의 도시 연비 t 검정

mpg_diff2 <- mpg %>%
  select(fl, cty) %>%
  filter(fl %in% c("r", "p")) # r:regular, p:premium

table(mpg_diff2$fl)

t.test(data = mpg_diff2, cty ~ fl, val.equal = T)

# data:  cty by fl
# t = 1.2118, df = 107.23, p-value = 0.2283
# p-value가 0.05보다 큰 0.2875이다.
# 실제로는 차이가 없는데 우연에 의해 이런 차이가 관찰될 확률이 28.75%라는 의미이다.
# 따라서 일반 휘발유와 고급 휘발유를 사용하는 자동차 간 도시 연비 차이가 통계적으로 유의하지 않다.

# alternative hypothesis: true difference in means is not equal to 0
# 95 percent confidence interval:
#   -0.3989031  1.6534818

# sample estimates:
#   mean in group p mean in group r 
# 17.36538        16.73810  # 고급 휘발유 자동차의 도시 연비 평균이 0.6 높지만
                            # 이런 차이는 우연이 발생했을 가능성이 크다고 해석



```
